def prepare_data(engineered=True, scale=True, log_target=True, split_type='chrono'):
    d = data.copy()
    d['StateHoliday'] = d['StateHoliday'].astype(str)

    if engineered:
        d['Year'] = d.Date.dt.year
        d['Month'] = d.Date.dt.month
        d['Day'] = d.Date.dt.day
        d['WeekOfYear'] = d.Date.dt.isocalendar().week.astype(int)

        median_dist = d['CompetitionDistance'].median()
        d['CompetitionDistance'] = np.log1p(d['CompetitionDistance'].fillna(median_dist))

        comp_months = (d['Year'] - d['CompetitionOpenSinceYear']) * 12 + \
                      (d['Month'] - d['CompetitionOpenSinceMonth'])
        d['CompetitionOpenMonths'] = comp_months.clip(lower=0).fillna(0)

        promo2_months = (d['Year'] - d['Promo2SinceYear']) * 12 + \
                        (d['WeekOfYear'] - d['Promo2SinceWeek']) / 4.35
        d['Promo2Months'] = (promo2_months.clip(lower=0) * d['Promo2']).fillna(0)

        month_map = {1:'Jan', 2:'Feb', 3:'Mar', 4:'Apr', 5:'May', 6:'Jun',
                     7:'Jul', 8:'Aug', 9:'Sept', 10:'Oct', 11:'Nov', 12:'Dec'}
        is_promo = []
        for pi, m in zip(d['PromoInterval'], d['Month']):
            if isinstance(pi, str) and month_map[m] in pi.split(','):
                is_promo.append(1)
            else:
                is_promo.append(0)
        d['IsPromoMonth'] = is_promo

        d = d.drop(columns=['CompetitionOpenSinceMonth', 'CompetitionOpenSinceYear',
                            'Promo2SinceWeek', 'Promo2SinceYear', 'PromoInterval'])
        d = pd.get_dummies(d, columns=['StateHoliday', 'StoreType', 'Assortment'], drop_first=True)
    else:
        d = pd.get_dummies(d, columns=['StateHoliday', 'StoreType', 'Assortment', 'PromoInterval'],
                           drop_first=True)

    y = d['Sales'].values.astype('float32')
    X = d.drop(columns=['Sales', 'Date']).fillna(0)
    feature_names = list(X.columns)
    X = X.values.astype('float32')

    n = len(X)
    if split_type == 'random':
        idx = np.random.RandomState(SEED).permutation(n)
        X = X[idx]
        y = y[idx]

    t1 = int(0.80 * n)
    t2 = int(0.90 * n)
    t3 = int(0.95 * n)

    result = {
        'X_train': X[:t1], 'X_val': X[t1:t2], 'X_test': X[t2:t3], 'X_hold': X[t3:],
        'y_train': y[:t1], 'y_val': y[t1:t2], 'y_test': y[t2:t3], 'y_hold': y[t3:],
        'features': feature_names, 'log_target': log_target
    }

    if scale:
        scaler = StandardScaler()
        scaler.fit(result['X_train'])
        for k in ['X_train', 'X_val', 'X_test', 'X_hold']:
            result[k] = scaler.transform(result[k]).astype('float32')

    return result

def build_model(n_features, hidden_layers, dropout_rates, learning_rate=0.001, optimizer_type='adam'):
    model = Sequential()
    model.add(Input(shape=(n_features,)))

    for i in range(len(hidden_layers)):
        model.add(Dense(hidden_layers[i], activation='relu'))
        if i < len(dropout_rates) and dropout_rates[i] > 0:
            model.add(Dropout(dropout_rates[i]))

    model.add(Dense(1))

    if optimizer_type == 'adam':
        opt = Adam(learning_rate=learning_rate)
    else:
        opt = SGD(learning_rate=learning_rate, momentum=0.9)

    model.compile(optimizer=opt, loss='mse', metrics=['mae'])
    return model


def get_predictions(model, X, y_true, log_target):
    raw_pred = model.predict(X, batch_size=8192, verbose=0).flatten()
    if log_target:
        pred_eur = np.clip(np.expm1(raw_pred), 0, None)
    else:
        pred_eur = np.clip(raw_pred, 0, None)
    mae = mean_absolute_error(y_true, pred_eur)
    rmse = math.sqrt(mean_squared_error(y_true, pred_eur))
    return mae, rmse, pred_eur


def run_experiment(exp_id, description, data_dict,
                   hidden_layers=[256, 128, 64],
                   dropout_rates=[0.3, 0.2, 0.0],
                   learning_rate=0.001,
                   optimizer_type='adam',
                   batch_size=512,
                   max_epochs=25,
                   use_early_stopping=True,
                   save_history_as=None):

    np.random.seed(SEED)
    tf.random.set_seed(SEED)
    tf.keras.backend.clear_session()

    n_features = data_dict['X_train'].shape[1]
    log_target = data_dict['log_target']

    model = build_model(n_features, hidden_layers, dropout_rates, learning_rate, optimizer_type)

    cbs = []
    if use_early_stopping:
        cbs.append(EarlyStopping(monitor='val_loss', patience=5, restore_best_weights=True))

    if log_target:
        y_tr = np.log1p(data_dict['y_train'])
        y_va = np.log1p(data_dict['y_val'])
    else:
        y_tr = data_dict['y_train']
        y_va = data_dict['y_val']

    start = time.time()
    history = model.fit(
        data_dict['X_train'], y_tr,
        validation_data=(data_dict['X_val'], y_va),
        epochs=max_epochs,
        batch_size=batch_size,
        verbose=0,
        callbacks=cbs
    )
    elapsed = time.time() - start

    val_mae, val_rmse, _ = get_predictions(model, data_dict['X_val'], data_dict['y_val'], log_target)

    result = {
        'id': exp_id,
        'desc': description,
        'val_mae': round(val_mae, 2),
        'val_rmse': round(val_rmse, 2),
        'epochs': len(history.history['loss']),
        'train_s': round(elapsed, 1)
    }

    with open('out/results.jsonl', 'a') as f:
        f.write(json.dumps(result) + '\n')

    print(f"[{exp_id}] {description}")
    print(f"     Val MAE: {val_mae:,.1f} | Val RMSE: {val_rmse:,.1f} | {result['epochs']} ep | {elapsed:.0f}s")

    if save_history_as:
        hist_dict = {}
        for k, v in history.history.items():
            hist_dict[k] = [float(x) for x in v]
        json.dump(hist_dict, open(f'out/hist_{save_history_as}.json', 'w'))

    return model
